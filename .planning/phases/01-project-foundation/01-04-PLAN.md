# Plan 01-04: Navigation Skeleton

## Objective

Wire up Compose Navigation with bottom nav bar and 3 placeholder screens. This is the structural skeleton all UI phases will build on.

## Navigation Design

- **Bottom nav:** 2 tabs — 🔔 **Alarms** (AlarmListScreen) and ⚙️ **Settings** (SettingsScreen)
- **AlarmSetterScreen:** No bottom tab — reached by FAB from AlarmList or tapping an alarm card
- All 3 screens start as placeholder composables with centered screen-name text

package com.biobell.android.ui.navigation

sealed class Screen(val route: String) {
    object AlarmList : Screen("alarm_list")
    object AlarmSetter : Screen("alarm_setter?alarmId={alarmId}") {

### Task 3: BottomNavBar.kt

```kotlin
package com.biobell.android.ui.navigation

import androidx.compose.material.icons.Icons
import androidx.compose.material.icons.filled.Alarm
import androidx.compose.material.icons.filled.Settings
import androidx.compose.material.icons.outlined.Alarm
import androidx.compose.material.icons.outlined.Settings
import androidx.compose.material3.*
import androidx.compose.runtime.Composable
import androidx.navigation.NavController
import androidx.navigation.compose.currentBackStackEntryAsState

val bottomNavItems = listOf(
    BottomNavItem(Screen.AlarmList, "Alarms", Icons.Filled.Alarm, Icons.Outlined.Alarm),
    BottomNavItem(Screen.Settings, "Settings", Icons.Filled.Settings, Icons.Outlined.Settings),
)

@Composable
fun BioBellBottomBar(navController: NavController) {
    val navBackStackEntry = navController.currentBackStackEntryAsState()
    val currentRoute = navBackStackEntry.value?.destination?.route

    NavigationBar {
        bottomNavItems.forEach { item ->
            val selected = currentRoute == item.screen.route
            NavigationBarItem(
                selected = selected,
                onClick = {
                    navController.navigate(item.screen.route) {
                        popUpTo(Screen.AlarmList.route) { saveState = true }
                        launchSingleTop = true
                        restoreState = true
                    }
                },
                icon = {
                    Icon(
                        imageVector = if (selected) item.selectedIcon else item.unselectedIcon,
                        contentDescription = item.label
                    )
                },
                label = { Text(item.label) }
            )
        }
    }
}
```

### Task 4: BioBellNavGraph.kt

```kotlin
package com.biobell.android.ui.navigation

import androidx.compose.runtime.Composable
import androidx.navigation.NavHostController
import androidx.navigation.NavType
import androidx.navigation.compose.NavHost
import androidx.navigation.compose.composable
import androidx.navigation.navArgument
import com.biobell.android.ui.alarm.AlarmListScreen
import com.biobell.android.ui.alarm.AlarmSetterScreen
import com.biobell.android.ui.settings.SettingsScreen

@Composable
fun BioBellNavGraph(navController: NavHostController) {
    NavHost(
        navController = navController,
        startDestination = Screen.AlarmList.route
    ) {
        composable(Screen.AlarmList.route) {
            AlarmListScreen(
                onCreateAlarm = { navController.navigate(Screen.AlarmSetter.createRoute()) },
                onEditAlarm = { id -> navController.navigate(Screen.AlarmSetter.createRoute(id)) }
            )
        }
        composable(
            route = Screen.AlarmSetter.route,
            arguments = listOf(navArgument("alarmId") {
                type = NavType.LongType
                defaultValue = -1L
            })
        ) { backStackEntry ->
            val alarmId = backStackEntry.arguments?.getLong("alarmId")?.takeIf { it != -1L }
            AlarmSetterScreen(
                alarmId = alarmId,
                onNavigateBack = { navController.popBackStack() }
            )
        }
        composable(Screen.Settings.route) {
            SettingsScreen()
        }
    }
}
```
